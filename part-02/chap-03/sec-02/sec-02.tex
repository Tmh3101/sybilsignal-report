\subsection{2. Kết quả thực nghiệm}

\indent\indent Toàn bộ chu trình thực nghiệm mô hình \textbf{SybilSignal} — từ khai thác dữ liệu qua BigQuery, huấn luyện GAE/GNN, đến đánh giá các mô hình — được thực hiện nhất quán trên nền tảng \textbf{Google Colab}. Chi tiết cấu hình phần cứng và các thư viện phần mềm phục vụ cho mục đích thực nghiệm và đánh giá được trình bày dưới đây:

\begin{itemize}[label={--}, itemsep=1pt]
    \item \textbf{Cấu hình phần cứng thực nghiệm (Google Colab):}
    \begin{itemize}[label={+}, itemsep=1pt]
        \item \textbf{Hệ điều hành:} Ubuntu 22.04.3 LTS.
        \item \textbf{GPU:} NVIDIA Tesla T4 (16 GiB GDDR6).
        \item \textbf{RAM:} 12.7 GiB System RAM.
        \item \textbf{CPU:} Intel(R) Xeon(R) CPU @ 2.20GHz.
    \end{itemize}
    
    \item \textbf{Hệ thống thư viện và công cụ:}
    \begin{itemize}[label={+}, itemsep=1pt]
        \item \textbf{Khai thác và xử lý dữ liệu:} \texttt{google-cloud-bigquery} để truy vấn dữ liệu từ Lens Protocol; \texttt{pandas} và \texttt{numpy} phục vụ lưu trữ và biến đổi ma trận đặc trưng.
        \item \textbf{Học sâu và phân tích đồ thị:} \texttt{torch} và \texttt{torch-geometric} để xây dựng các kiến trúc GNN; \texttt{sentence-transformers} và \texttt{transformers} cho quy trình nhúng ngữ nghĩa văn bản.
        \item \textbf{Đánh giá và Lưu trữ mô hình:} \texttt{scikit-learn} tính toán các thang đo hiệu năng (Macro F1, AUC-ROC);
    \end{itemize}
\end{itemize}
