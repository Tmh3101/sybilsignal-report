\subsubsection{1.1. Phân hệ học biểu diễn và gán nhãn}

\indent\indent Quy trình của phân hệ này được minh họa tại \textbf{Hình \ref{fig:DataPipeline}}, tập trung vào việc học biểu diễn không gian tiềm ẩn, tự động hóa khâu gán nhãn nhằm giải quyết sự thiếu hụt dữ liệu huấn luyện.

\begin{figure}[H]
    \centering
    \includegraphics[width=\linewidth]{images/part-02/chap-02/sec-01/DataPipeline.png}
    \vspace{0.1cm}
    \caption{Sơ đồ kiến trúc luồng học biểu diễn và gán nhãn}
    \label{fig:DataPipeline}

\textbf{Đặc tả dữ liệu:}
\begin{itemize}[label={--}, itemsep=1pt]
    \item \textbf{Đầu vào:} Một khoảng thời gian cụ thể. Hệ thống sẽ dựa vào khoảng thời gian này để truy xuất dữ liệu lịch sử hàng loạt từ Lens Protocol thông qua BigQuery (bao gồm thông tin các tài khoản, số liệu thống kê và các bản ghi tương tác).
    \item \textbf{Đầu ra:} Một tập dữ liệu đồ thị có nhãn, sẵn sàng làm nguyên liệu để huấn luyện các mô hình phân lớp.
\end{itemize}\vspace{0.2cm}

Kiến trúc tại được thiết kế nhằm tối ưu hóa sự luân chuyển thông tin qua 04 phân vùng chức năng cốt lõi:
\begin{itemize}[label={--}, itemsep=1pt]
    \item \textbf{Luồng tích hợp đa nguồn:} Dữ liệu từ Lens Protocol được xử lý song song để tạo ra vector đặc trưng nút $396$ chiều. Thiết kế này đảm bảo sự hội tụ giữa ngữ nghĩa văn bản và hành vi định lượng trước khi bước vào không gian đồ thị.
    \item \textbf{Luồng nén và trích xuất đặc trưng:} Kế thừa cơ chế Encoder-Decoder của GAE tại Chương 1, đồ thị đa tầng được đưa qua mạng GATv2 để nén cấu trúc. Quá trình này giúp lọc nhiễu và trích xuất các đặc trưng cấu trúc cô đọng nhất dưới dạng các vector tiềm ẩn.
    \item \textbf{Luồng gán nhãn giả lập:} Dựa trên các vector đã nén, hệ thống thực hiện phân tách không gian bằng K-Means. Điểm nhấn trong thiết kế này là sự kết hợp với hệ luật chuyên gia để ánh xạ điểm rủi ro thành nhãn, hoàn tất quy trình tự động hóa khâu chuẩn bị dữ liệu giám sát mà không cần can thiệp thủ công.
\end{itemize}

\textit{1.1.2. Kiến trúc GAE + GATv2}\vspace{0.2cm}

Với cơ sở toán học của bộ mã hóa đồ thị đã được thiết lập tại Mục 2.4 (Chương 1), phần này đi sâu vào việc cấu hình tham số mạng thực tế nhằm tối ưu hóa khả năng nén đặc trưng cấu trúc. Để thực hiện hạ chiều từ vector đặc trưng nút 396 chiều gốc, bộ mã hóa được thiết kế với cấu trúc GATv2 hai tầng ẩn. Quá trình này thực hiện hạ chiều dữ liệu tuần tự về 32 chiều và cuối cùng là 16 chiều trong không gian tiềm ẩn, đảm bảo sự cân bằng giữa tính cô đọng dữ liệu và tính bảo toàn các ràng buộc đồ thị đa quan hệ.\vspace{0.3cm}

Kiến trúc chi tiết của mô hình GAE được minh họa tại \textbf{Hình \ref{fig:GAEPipeline}}. Mô hình được thiết kế với hai thành phần cốt lõi: Bộ mã hóa và Bộ giải mã.

\begin{figure}[H]
    \centering
    \includegraphics[width=0.5\linewidth]{images/part-02/chap-02/sec-01/GAEPipeline.png}
    \vspace{0.2cm}
    \caption{Kiến trúc Mạng tự mã hóa đồ thị với bộ mã hóa GATv2}

\end{figure}

Hệ thống sử dụng Mạng nơ-ron tích chập đồ thị có cơ chế chú ý nâng cao (\textit{GATv2}) làm bộ mã hóa. Đặc biệt, bộ mã hóa được thiết kế để duy trì cơ chế Điều hướng chú ý thông qua tham số \texttt{edge\_dim=1}, giúp hấp thụ trực tiếp các trọng số cạnh đã được chuẩn hóa Logarit ở giai đoạn trước vào mọi bước nhảy lan truyền.\vspace{0.3cm}

Chi tiết luồng xử lý qua các tầng của bộ mã hóa được minh họa tại \textbf{Hình \ref{fig:GATEncoder}}:
\begin{figure}[H]
    \centering
    % Đổi tên file ảnh dưới đây cho khớp với sơ đồ GAT Encoder bạn vừa vẽ
    \includegraphics[width=0.5\linewidth]{images/part-02/chap-02/sec-01/GATEncoder.png}
    \vspace{0.2cm}
    \caption{Kiến trúc luồng dữ liệu bên trong Bộ mã hóa GATv2}

Kiến trúc mã hóa bao gồm hai tầng tích chập:
\begin{itemize}[label={--}, itemsep=3pt]
    \item \textbf{Tầng tích chập thứ nhất (Conv1):} Nhận đầu vào là ma trận đặc trưng $X \in \mathbb{R}^{N \times 396}$. Tầng này sử dụng cơ chế đa chú ý với $4$ heads nhằm khai phá đặc trưng dưới nhiều không gian biểu diễn khác nhau. Mỗi head ánh xạ dữ liệu thành không gian $32$ chiều. Các kết quả sau đó được ghép nối lại, tạo ra vector trung gian có kích thước $128$ chiều ($32 \times 4$). Đầu ra đi qua hàm kích hoạt phi tuyến ELU và một lớp Dropout với tỷ lệ $0.1$ nhằm ngăn chặn over-fitting.
    \item \textbf{Tầng tích chập thứ hai (Conv2):} Nhận vector $128$ chiều làm đầu vào và thiết lập cơ chế chú ý đơn ($1$ head). Cấu hình này ép buộc mô hình tổng hợp các góc nhìn từ tầng trước để nén thông tin thành vector tiềm ẩn cuối cùng $Z \in \mathbb{R}^{N \times 16}$.
\end{itemize}

Thay vì sử dụng một mạng nơ-ron phức tạp, bộ giải mã của GAE tính toán tích vô hướng giữa các vector embedding của các cặp nút ($Z_i \cdot Z_j$) để dự đoán xác suất tồn tại của một cạnh nối giữa chúng. Đầu ra được đánh giá thông qua hàm lỗi tái tạo dựa trên Binary Cross-Entropy (BCE). GAE sẽ liên tục sử dụng sai số tái tạo để lan truyền ngược, ép các trọng số của GATv2 phải điều chỉnh sao cho đồ thị tái tạo từ không gian tiềm ẩn gần giống nhất với đồ thị đa quan hệ gốc ban đầu.\vspace{0.3cm}

\textbf{Đầu ra cuối cùng:} Sau khi kết thúc quá trình tối ưu hóa, mô hình trích xuất ma trận có kích thước $N \times 16$. Ma trận này chứa đựng toàn bộ tri thức về cấu trúc mạng lưới của đồ thị, đóng vai trò là đầu vào tiên quyết cho quy trình phân cụm và gán nhãn giả lập phía sau.\vspace{0.3cm}

\textbf{* Tối ưu hóa phân cụm K-Means}\vspace{0.2cm}

\textbf{* Cơ chế phân lớp mờ}\vspace{0.2cm}

Kế thừa bản chất định nghĩa của 04 nhóm rủi ro đã được phân tích lý thuyết tại Mục 2.5 (Chương 1), phân hệ này tiến hành cài đặt hàm phân ngưỡng mờ. Điểm đặc thù của mạng lưới Web3 là ranh giới giữa người dùng thật và các tài khoản Sybil tinh vi thường không rõ ràng, tạo ra các vùng giao thoa hành vi. \vspace{0.3cm}

Để giải quyết vấn đề này, điểm rủi ro liên tục ($0-100$) tổng hợp từ hệ luật chuyên gia được phân phối qua các hàm liên thuộc và ánh xạ cứng thành các nhãn rời rạc. Cấu hình tham số cắt ngưỡng thực tế được minh họa trực quan tại \textbf{Hình \ref{fig:FuzzyDistribution}} và liệt kê cụ thể như sau:

\begin{figure}[H]
    \centering
    \includegraphics[width=0.75\linewidth]{images/part-02/chap-02/sec-01/FuzzyLabelingDistribution.png}
    \caption{Trực quan hóa sự giao thoa hành vi và các ngưỡng phân lớp mờ}
    \vspace{0.1cm}

\begin{itemize}[label={--}, itemsep=1pt]
    \item \texttt{Risk Score} $\leq 20$: Phân loại \texttt{BENIGN} (Vùng an toàn, đỉnh phân phối của người dùng thật).
    \item $20 <$ \texttt{Risk Score} $\leq 50$: Phân loại \texttt{LOW\_RISK} (Vùng đáng ngờ nhẹ, độ giao thoa thấp).
    \item $50 <$ \texttt{Risk Score} $\leq 80$: Phân loại \texttt{HIGH\_RISK} (Vùng rủi ro cao, hành vi ngụy trang của lính ảo).
    \item \texttt{Risk Score} $> 80$: Phân loại \texttt{MALICIOUS} (Vùng vi phạm rõ rệt của mạng lưới Sybil).
\end{itemize}
