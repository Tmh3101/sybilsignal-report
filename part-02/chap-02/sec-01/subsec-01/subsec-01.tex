\subsubsection{1.1. Quy trình xây dựng dữ liệu đồ thị đa tầng nhận thức ràng buộc}

\indent\indent Trước khi đưa vào huấn luyện các mô hình học sâu đồ thị, dữ liệu thô từ mạng xã hội phi tập trung cần trải qua một quy trình tiền xử lý phức tạp. Mục tiêu của giai đoạn này là chuyển đổi các bản ghi cơ sở dữ liệu phẳng thành một cấu trúc đồ thị đa quan hệ có hướng, trong đó mỗi nút và cạnh đều mang lượng thông tin ngữ nghĩa và toán học cao nhất.\vspace{0.3cm}

\indent Sơ đồ luồng xử lý thu thập và kiến tạo đồ thị được minh họa tại \textbf{Hình \ref{fig:GraphBuilder}}, bao gồm hai nhánh xử lý song song và cơ chế hoạt động như sau:
\begin{figure}[H]
    \centering
    \includegraphics[width=\linewidth]{images/part-02/chap-02/sec-01/BuildMLG.png}
    \vspace{0.1cm}
    \caption{Sơ đồ luồng quy trình xây dựng dữ liệu đồ thị đa tầng}
    \label{fig:GraphBuilder}
\end{figure}

\textbf{Đầu vào:} Dữ liệu thô từ mạng lưới Lens Protocol được truy xuất hàng loạt thông qua BigQuery, bao gồm thông tin hồ sơ, thống kê hoạt động và các bản ghi sự kiện tương tác.\vspace{0.3cm}

\textbf{Đầu ra:} Một đồ thị đa quan hệ hoàn chỉnh, được đóng gói dưới dạng đối tượng \texttt{Data} của thư viện PyTorch Geometric.\vspace{0.3cm}

\textbf{Quy trình xử lý:}\vspace{-0.1cm}
\begin{itemize}[label={--}, itemsep=3pt]
    \item \textbf{Thu thập và xử lý dữ liệu nút:} Dữ liệu mô tả tài khoản được chia làm hai nhánh trích xuất đặc trưng độc lập trước khi tiến hành hợp nhất:
    \begin{itemize}[label={+}, itemsep=3pt]
        \item \textit{Đặc trưng ngữ nghĩa:} Nhúng qua mô hình \texttt{all-MiniLM-L6-v2} tạo vector 384 chiều.
        \item \textit{Đặc trưng định lượng (12 chiều):} Gồm 11 chỉ số tương tác và 01 chỉ số thời gian hoạt động, được chuẩn hóa qua \textit{StandardScaler}.
        \item \textit{Hợp nhất:} Tạo thành vector đặc trưng nút toàn diện với kích thước 396 chiều.
    \end{itemize}

    \item \textbf{Thu thập và xử lý dữ liệu quan hệ:} Kế thừa lý thuyết về Đồ thị đa quan hệ nhận thức ràng buộc đã được định nghĩa tại Mục 2.2 (Chương 1). Để tối ưu hóa không gian biểu diễn và khắc phục vấn đề phân mảnh dữ liệu, toàn bộ 14 loại quan hệ tương tác gốc từ dữ liệu chuỗi khối được hệ thống phân tích, ánh xạ và hợp nhất về mặt ngữ nghĩa thành 04 tầng đồ thị chính. Bảng \ref{tab:edge_mapping} trình bày chi tiết quy trình gom nhóm này:

    \begin{table}[H]
        \centering
        \caption{Bảng ánh xạ 14 loại quan hệ gốc thành 4 tầng đồ thị ngữ nghĩa}
        \label{tab:edge_mapping}
        \renewcommand{\arraystretch}{1.4} % Tăng độ thoáng cho dòng để không bị dính sát đường kẻ
        \setlength{\tabcolsep}{8pt}
        \begin{tabular}{|p{3cm}|p{8cm}|c|}
            \hline
            \textbf{Tầng đồ thị} & \textbf{Các loại quan hệ gốc} & \textbf{Số lượng} \\

            \hline
            \textbf{2. Interact Layer} & \texttt{COMMENT}, \texttt{COMMENT\_REV}, \texttt{QUOTE}, \texttt{QUOTE\_REV}, \texttt{UPVOTE}, \texttt{UPVOTE\_REV}, \texttt{COLLECT}, \texttt{COLLECT\_REV}, \texttt{TIP}, \texttt{TIP\_REV} & 10 \\

        \end{tabular}

    Mặc dù được phân nhóm dưới góc độ 04 tầng ngữ nghĩa để xây dựng cơ sở lý luận, hệ thống thực tế vẫn bảo toàn nguyên vẹn sự đa dạng của 14 loại quan hệ thông qua chỉ mục phân loại cạnh. Việc giữ độc lập các loại cạnh thay vì gộp chung giúp mạng nơ-ron đồ thị khai thác được đặc trưng vi mô của từng hành vi đơn lẻ. Quá trình xử lý cạnh được tiếp tục qua các bước kỹ thuật:

    \begin{itemize}[label={+}, itemsep=1pt]
        \item \textit{Sinh cạnh ngược:} Đối với các loại quan hệ có hướng (nhóm Theo dõi và Tương tác), hệ thống sinh thêm các chiều cạnh ngược tương ứng (ví dụ: \texttt{FOLLOW\_REV}) để đảm bảo thông điệp lan truyền linh hoạt theo cả hai chiều. Riêng các cạnh \texttt{SIMILARITY} và \texttt{CO-OWNER} vốn mang tính đối xứng nên được giữ nguyên cấu trúc vô hướng.
        \item \textit{Gán trọng số cạnh:} Với mỗi loại cạnh cụ thể giữa hai thực thể, hệ thống gán trọng số cơ sở ($W_{base}$) và áp dụng hàm chuẩn hóa Logarit trên tần suất xuất hiện của hành vi đó. Cơ chế này giúp kiểm soát cường độ tương tác, hạn chế nhiễu.
    \end{itemize}
    
    \item \textbf{Đóng gói thành Đồ thị đa quan hệ:} Các định danh tài khoản được ánh xạ tuần tự thành các chỉ mục số nguyên. Cuối cùng, toàn bộ dữ liệu được đóng gói thành đối tượng đồ thị chuẩn của thư viện PyTorch Geometric, bao gồm: ma trận đặc trưng nút \texttt{x}, ma trận liên kết \texttt{edge\_index}, vector trọng số \texttt{edge\_attr} và vector chỉ mục phân loại 14 loại cạnh \texttt{edge\_type}.
\end{itemize}
